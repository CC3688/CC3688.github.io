# 数组

# reduce

const arr = [
  {id: 1, num: 1, active: true},
  {id: 2, num: 2, active: false},
  {id: 3, num: 3, active: true},
]

// 用 reduce 得到一个 总数
const res = arr.reduce((sum, c)=>{
  if(c.active) {
    sum += c.num
  }
  return sum
}, 0)

// 注意,  reduce的第二个参数会作为 sum 的初始值, 数组的第一个元素是第一次迭代的值 
// 不传第二个参数是,则 第一次迭代时, sum 是数组的第一个元素, c是第二个元素

# 利用 ref 调用组件的方法和属性

// 组件acomp 里有个name: 'xxx'  及 方法  show(){}
// 在使用 acomp的组件里可能通过 ref 使用acomp的 属性和方法
<acomp ref="acomp"><acomp>

// 在使用acomp的 methods 
test() {
  this.$refs.acomp.show()   //使用 该方法,可以参传
  this.$refs.acomp.name     // 使用 属性
}
